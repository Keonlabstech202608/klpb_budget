/** * KEONLABS - Business Logic Layer * * This module contains all business logic independent of UI and database. * It handles calculations, validations, and data transformations. * * Business logic is completely separated from: * - Database operations (use dbService) * - UI rendering (use ui-engine) * - Authentication (use authService) */ class BusinessLogic { constructor() { this.months = ["January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"]; } /** * Validate stock item data * @param {object} item - Stock item to validate * @returns {object} - Validation result {isValid: boolean, errors: array} */ validateStockItem(item) { const errors = []; if (!item.name || item.name.trim() === '') { errors.push('Item name is required'); } if (item.totalCapitalInvested === undefined || item.totalCapitalInvested < 0) { errors.push('Capital invested must be a positive number'); } if (item.currentStock === undefined || item.currentStock < 0) { errors.push('Stock quantity must be a positive number'); } return { isValid: errors.length === 0, errors: errors }; } /** * Calculate low stock warning * @param {number} currentStock - Current stock quantity * @param {number} threshold - Low stock threshold (default: 3) * @returns {boolean} - True if stock is low */ isStockLow(currentStock, threshold = 3) { return currentStock <= threshold; } /** * Calculate monthly statistics * @param {array} businessData - All business data * @param {array} stockHistory - Stock history * @param {array} globalLedger - Global ledger * @param {number} month - Month (0-11) * @param {number} year - Year * @returns {object} - Monthly stats */ calculateMonthlyStats(businessData, stockHistory, globalLedger, month, year) { const monthlyCapital = stockHistory .filter(x => x.month === month && x.year === year) .reduce((sum, x) => sum + (parseFloat(x.cost) || 0), 0); const monthlyRevenue = businessData .filter(x => x.month === month && x.year === year) .reduce((sum, x) => sum + (parseFloat(x.totalSalesRevenue) || 0), 0); const monthlyWithdrawals = globalLedger .filter(x => x.month === month && x.year === year && x.type === 'WITHDRAW') .reduce((sum, x) => sum + x.amt, 0); const netProfit = monthlyRevenue - monthlyCapital - monthlyWithdrawals; return { capital: monthlyCapital, revenue: monthlyRevenue, withdrawals: monthlyWithdrawals, profit: netProfit }; } /** * Calculate total debtor balance * @param {array} debtors - Array of debtors * @returns {number} - Total balance */ calculateTotalDebtorBalance(debtors) { return debtors.reduce((sum, debtor) => sum + debtor.balance, 0); } /** * Check if debt is overdue * @param {string} debtDate - Debt creation date * @param {number} daysThreshold - Days before considered overdue (default: 5) * @returns {boolean} - True if overdue */ isDebtOverdue(debtDate, daysThreshold = 5) { const daysDifference = Math.floor( Math.abs(new Date() - new Date(debtDate)) / (1000 * 60 * 60 * 24) ); return daysDifference >= daysThreshold; } /** * Check if date is within this week * @param {string} dateString - Date string * @returns {boolean} - True if within this week */ isThisWeek(dateString) { const daysDifference = Math.ceil( Math.abs(new Date() - new Date(dateString)) / (1000 * 60 * 60 * 24) ); return daysDifference <= 7; } /** * Filter ledger entries by criteria * @param {array} ledger - Global ledger * @param {string} type - Filter by type ('ALL', 'SALE', 'INCOME', etc.) * @param {string} searchTerm - Search in description * @param {string} timeFrame - Time frame ('YEAR', 'MONTH', 'WEEK') * @param {number} month - Month (0-11) * @param {number} year - Year * @returns {array} - Filtered ledger */ filterLedger(ledger, type, searchTerm, timeFrame, month, year) { return ledger.filter(entry => { const typeMatch = type === 'ALL' || entry.type === type; const searchMatch = entry.desc.toLowerCase().includes(searchTerm.toLowerCase()); const timeMatch = (timeFrame === 'YEAR' && entry.year === year) || (timeFrame === 'MONTH' && entry.year === year && entry.month === month) || (timeFrame === 'WEEK' && this.isThisWeek(entry.date)); return typeMatch && searchMatch && timeMatch; }); } /** * Calculate ledger summary * @param {array} ledger - Filtered ledger entries * @returns {object} - Summary {income, expense, net} */ calculateLedgerSummary(ledger) { let totalIncome = 0; let totalExpense = 0; ledger.forEach(entry => { const amount = Math.abs(entry.amt); if (['SALE', 'INCOME'].includes(entry.type)) { totalIncome += amount; } else { totalExpense += amount; } }); return { income: totalIncome, expense: totalExpense, net: totalIncome - totalExpense }; } /** * Calculate personal budget summary * @param {array} personalLedger - Personal ledger entries * @param {number} month - Month (0-11) * @param {number} year - Year * @returns {object} - Summary {income, expense, net} */ calculateBudgetSummary(personalLedger, month, year) { const monthlyEntries = personalLedger.filter(x => x.month === month && x.year === year); const income = monthlyEntries .filter(x => x.type === 'income') .reduce((sum, x) => sum + x.amt, 0); const expense = monthlyEntries .filter(x => x.type === 'expense') .reduce((sum, x) => sum + x.amt, 0); return { income: income, expense: expense, net: income - expense }; } /** * Validate sale transaction * @param {object} item - Stock item * @param {number} quantity - Quantity to sell * @returns {object} - Validation result {isValid: boolean, error: string} */ validateSaleTransaction(item, quantity) { if (!item) { return { isValid: false, error: 'Item not found' }; } if (quantity <= 0) { return { isValid: false, error: 'Quantity must be greater than 0' }; } if (item.currentStock < quantity) { return { isValid: false, error: `Insufficient stock. Available: ${item.currentStock}` }; } return { isValid: true }; } /** * Calculate sale amount * @param {number} unitPrice - Unit sell price * @param {number} quantity - Quantity sold * @returns {number} - Total sale amount */ calculateSaleAmount(unitPrice, quantity) { return unitPrice * quantity; } /** * Get month name * @param {number} monthIndex - Month index (0-11) * @returns {string} - Month name */ getMonthName(monthIndex) { return this.months[monthIndex] || ''; } /** * Format currency * @param {number} amount - Amount to format * @param {string} currency - Currency code (default: 'K') * @returns {string} - Formatted currency */ formatCurrency(amount, currency = 'K') { return `${currency} ${amount.toFixed(2)}`; } } // Create and export the business logic instance const businessLogic = new BusinessLogic(); if (typeof module !== 'undefined' && module.exports) { module.exports = { BusinessLogic, businessLogic }; }
